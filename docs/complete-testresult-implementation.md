```javascript
import mongoose from 'mongoose';

const COLLECTION_NAME = 'test_results';

// ═══════════════════════════════════════════════════════════════════════════════
// TEST RESULT TYPES - Supported exam types
// ═══════════════════════════════════════════════════════════════════════════════
const TEST_RESULT_TYPES = {
    DIABETES_TEST: 'DIABETES_TEST',
    DNA_TEST: 'DNA_TEST',
    // More types can be added here
};

// ═══════════════════════════════════════════════════════════════════════════════
// TEST RESULT SCHEMA
// ═══════════════════════════════════════════════════════════════════════════════
/**
 * TestResult = AI Analysis Layer for Lab Results
 * 
 * ARCHITECTURE:
 * - Store rawData here (exam values from lab machine)
 * - Store aiAnalysis (AI suggestions based on raw data)
 * - Reference back to LabOrder (via labOrderId)
 * - Reference to MedicalRecord (for associating analysis with patient history)
 * 
 * DATA FLOW:
 * 1. Lab tech posts result → LabOrder.labResultData + TestResult.rawData
 * 2. AI system analyzes → TestResult.aiAnalysis updated
 * 3. Doctor reviews → Sees both raw data + AI suggestions
 * 4. Doctor confirms diagnosis → LabOrder.clinicalInterpretation
 * 
 * RELATIONSHIPS:
 * LabOrder ──1:1──> TestResult
 * MedicalRecord ──1:N──> TestResult (one record can have multiple tests)
 * Patient ──1:N──> TestResult (one patient can have many test results)
 */
const testResultSchema = new mongoose.Schema(
    {
        // ─────────────────────────────────────────────────────────────────
        // RELATIONSHIPS
        // ─────────────────────────────────────────────────────────────────

        labOrderId: {
            type: mongoose.Schema.Types.ObjectId,
            ref: 'lab_orders',
            description: 'Link to LabOrder (where raw data comes from)',
            index: true,
        },

        medicalRecordId: {
            type: mongoose.Schema.Types.ObjectId,
            ref: 'medical_records',
            required: true,
            description: 'Link to MedicalRecord (patient history context)',
            index: true,
        },

        patientId: {
            type: mongoose.Schema.Types.ObjectId,
            ref: 'patients',
            required: true,
            index: true,
        },

        createdBy: {
            type: mongoose.Schema.Types.ObjectId,
            ref: 'users',
            required: true,
            description: 'Lab tech or doctor who created result',
        },

        // ─────────────────────────────────────────────────────────────────
        // TEST METADATA
        // ─────────────────────────────────────────────────────────────────

        testType: {
            type: String,
            enum: Object.values(TEST_RESULT_TYPES),
            required: false,
            description: 'Type of test (DIABETES_TEST, DNA_TEST, etc.)',
        },

        // ─────────────────────────────────────────────────────────────────
        // RAW DATA (from lab machine/analyzer)
        // ─────────────────────────────────────────────────────────────────

        rawData: {
            type: mongoose.Schema.Types.Mixed,
            description: 'Raw exam values from laboratory machine',
            example: {
                glucose: 120,
                bloodPressure: '120/80',
                bmi: 28,
                hba1c: 6.8,
                // More fields based on test type
            },
        },

        // ─────────────────────────────────────────────────────────────────
        // AI ANALYSIS (generated by FastAPI service)
        // ─────────────────────────────────────────────────────────────────

        aiAnalysis: {
            type: mongoose.Schema.Types.Mixed,
            description: 'AI-generated analysis and suggestions',
            example: {
                diabetes: {
                    riskScore: 0.85,           // 0-1 scale
                    category: 'HIGH_RISK',     // LOW, MEDIUM, HIGH, CRITICAL
                    probability: 85,           // Percentage
                    recommendation: 'Start metformin',
                    details: {
                        hba1c: 6.8,
                        fasting: 120,
                        baseline: 'normal',
                        prediction: 'Type 2 Diabetes in 2 years',
                    },
                },
            },
        },

        // ─────────────────────────────────────────────────────────────────
        // SOFT DELETE
        // ─────────────────────────────────────────────────────────────────

        _destroy: {
            type: Boolean,
            default: false,
            index: true,
            description: 'Soft delete flag (GDPR compliance)',
        },
    },
    {
        timestamps: { createdAt: true, updatedAt: false },
        versionKey: false,
    }
);

// ═══════════════════════════════════════════════════════════════════════════════
// INDEXES
// ═══════════════════════════════════════════════════════════════════════════════

testResultSchema.index({ labOrderId: 1, _destroy: 1 });          // Find results by lab order
testResultSchema.index({ medicalRecordId: 1, _destroy: 1 });     // Find results by record
testResultSchema.index({ patientId: 1, createdAt: -1 });         // Patient's latest results
testResultSchema.index({ testType: 1, createdAt: -1 });          // Group by test type

// ═══════════════════════════════════════════════════════════════════════════════
// MODEL & EXPORTS
// ═══════════════════════════════════════════════════════════════════════════════

const TestResultModel = mongoose.model(COLLECTION_NAME, testResultSchema);

/**
 * Create new test result
 * @param {Object} data - Test result data
 * @returns {Object} Created TestResult document
 */
const createNew = async (data) => {
    return await TestResultModel.create(data);
};

/**
 * Find test result by ID
 * @param {String} id - TestResult ID
 * @returns {Object} TestResult document
 */
const findOneById = async (id) => {
    return await TestResultModel.findOne({ _id: id, _destroy: false });
};

/**
 * Find all test results for a lab order
 * @param {String} labOrderId - LabOrder ID
 * @returns {Array} TestResult documents
 */
const findByLabOrderId = async (labOrderId) => {
    return await TestResultModel.findOne({ 
        labOrderId, 
        _destroy: false 
    });
};

/**
 * Find all test results for a medical record
 * @param {String} medicalRecordId - MedicalRecord ID
 * @returns {Array} TestResult documents
 */
const findByMedicalRecordId = async (medicalRecordId) => {
    return await TestResultModel.find({ 
        medicalRecordId, 
        _destroy: false 
    }).sort({ createdAt: -1 });
};

/**
 * Find all test results for a patient
 * @param {String} patientId - Patient ID
 * @returns {Array} TestResult documents
 */
const findByPatientId = async (patientId) => {
    return await TestResultModel.find({ 
        patientId, 
        _destroy: false 
    }).sort({ createdAt: -1 });
};

/**
 * Update AI analysis for a test result
 * @param {String} id - TestResult ID
 * @param {Object} aiAnalysis - AI analysis data
 * @returns {Object} Updated TestResult
 */
const updateAiAnalysis = async (id, aiAnalysis) => {
    return await TestResultModel.findByIdAndUpdate(
        id,
        { aiAnalysis },
        { new: true, runValidators: true }
    );
};

export const testResultModel = {
    TestResultModel,
    TEST_RESULT_TYPES,
    createNew,
    findOneById,
    findByLabOrderId,
    findByMedicalRecordId,
    findByPatientId,
    updateAiAnalysis,
};
```

## 🔧 Service Layer

**File**: `src/services/testResult.service.js`

```javascript
import { medicalRecordModel } from '~/models/medicalRecord.model';
import { testResultModel } from '~/models/testResult.model';
import { labOrderModel } from '~/models/labOrder.model';
import ApiError from '~/utils/ApiError';
import { StatusCodes } from 'http-status-codes';
import { auditLogModel } from '~/models/auditLog.model';
import { patientModel } from '~/models/patient.model';
import { userModel } from '~/models/user.model';

// ═══════════════════════════════════════════════════════════════════════════════
// CONSTANTS
// ═══════════════════════════════════════════════════════════════════════════════

const AI_SERVICE_URL = process.env.AI_SERVICE_URL || 'http://localhost:8000/predict';

// ═══════════════════════════════════════════════════════════════════════════════
// SERVICE: CREATE NEW TEST RESULT (with optional AI analysis)
// ═══════════════════════════════════════════════════════════════════════════════

/**
 * Create new test result with optional AI analysis
 * 
 * FLOW:
 * 1. Validate medical record (must exist and be in CREATED state)
 * 2. If DIABETES_TEST → Call AI service for analysis
 * 3. Create TestResult with rawData + aiAnalysis (if available)
 * 4. Update MedicalRecord status to HAS_RESULT
 * 5. Create audit log
 * 
 * @param {String} labOrderId - LabOrder ID
 * @param {Object} body - Request body {testType, rawData}
 * @param {Object} currentUser - Current user (lab tech)
 * @returns {Object} Created TestResult
 */
const createNew = async (labOrderId, body, currentUser) => {
    const { testType, rawData } = body;

    // ─────────────────────────────────────────────────────────────────
    // STEP 1: Validate LabOrder exists
    // ─────────────────────────────────────────────────────────────────

    const labOrder = await labOrderModel.LabOrderModel.findById(labOrderId);
    if (!labOrder) {
        throw new ApiError(
            StatusCodes.NOT_FOUND,
            'Không tìm thấy lab order'
        );
    }

    // Verify lab order is in correct state for posting result
    if (labOrder.sampleStatus !== 'IN_PROGRESS') {
        throw new ApiError(
            StatusCodes.BAD_REQUEST,
            `Lab order must be IN_PROGRESS, currently: ${labOrder.sampleStatus}`
        );
    }

    // ─────────────────────────────────────────────────────────────────
    // STEP 2: Get medical record context
    // ─────────────────────────────────────────────────────────────────

    const medicalRecord = await medicalRecordModel.MedicalRecordModel.findById(
        labOrder.relatedMedicalRecordId
    );
    if (!medicalRecord) {
        throw new ApiError(
            StatusCodes.NOT_FOUND,
            'Không tìm thấy hồ sơ bệnh án liên quan'
        );
    }

    // ─────────────────────────────────────────────────────────────────
    // STEP 3: Get patient for AI analysis (age needed)
    // ─────────────────────────────────────────────────────────────────

    const patient = await patientModel.findById(labOrder.patientId);
    if (!patient) {
        throw new ApiError(
            StatusCodes.NOT_FOUND,
            'Không tìm thấy bệnh nhân'
        );
    }

    // ─────────────────────────────────────────────────────────────────
    // STEP 4: OPTIONAL - Call AI service for diabetes analysis
    // ─────────────────────────────────────────────────────────────────

    let aiAnalysis = {};

    if (testType === 'DIABETES_TEST') {
        try {
            const age = new Date().getFullYear() - patient.birthYear;

            console.log(`[TestResult] 🤖 Calling AI service for diabetes analysis...`);

            const aiResponse = await fetch(AI_SERVICE_URL, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify({
                    Pregnancies: rawData.pregnancies || 0,
                    Glucose: rawData.glucose,
                    BloodPressure: rawData.bloodPressure,
                    SkinThickness: rawData.skinThickness || 0,
                    Insulin: rawData.insulin || 0,
                    BMI: rawData.bmi,
                    DiabetesPedigreeFunction: rawData.diabetesPedigreeFunction || 0.5,
                    Age: age,
                }),
            });

            if (!aiResponse.ok) {
                throw new ApiError(
                    StatusCodes.SERVICE_UNAVAILABLE,
                    `AI service error: ${aiResponse.statusText}`
                );
            }

            const aiResult = await aiResponse.json();

            // Transform AI response to standard format
            aiAnalysis = {
                diabetes: {
                    riskScore: aiResult.probability,                    // 0-1
                    probability: Math.round(aiResult.probability * 100), // Percentage
                    category: aiResult.probability >= 0.8 ? 'HIGH_RISK' : 
                             aiResult.probability >= 0.6 ? 'MEDIUM_RISK' : 'LOW_RISK',
                    recommendation: generateRecommendation(aiResult.probability),
                    details: {
                        hba1c: rawData.hba1c,
                        fasting: rawData.glucose,
                        prediction: aiResult.risk,
                    },
                },
            };

            console.log(`[TestResult] ✅ AI analysis complete:`, aiAnalysis);
        } catch (aiError) {
            // Non-blocking: Log but don't fail if AI unavailable
            console.warn(
                `[TestResult] ⚠️ AI analysis failed (non-blocking):`,
                aiError.message
            );
            // Continue with just rawData
        }
    }

    // ─────────────────────────────────────────────────────────────────
    // STEP 5: Create TestResult with rawData + aiAnalysis
    // ─────────────────────────────────────────────────────────────────

    const testResultData = {
        labOrderId,
        medicalRecordId: labOrder.relatedMedicalRecordId,
        patientId: labOrder.patientId,
        createdBy: currentUser._id,
        testType,
        rawData,  // ✅ Store raw exam values
        aiAnalysis,  // ✅ Store AI suggestions (if available)
    };

    const testResult = await testResultModel.createNew(testResultData);

    if (!testResult) {
        throw new ApiError(
            StatusCodes.BAD_REQUEST,
            'Tạo TestResult thất bại'
        );
    }

    console.log(`[TestResult] ✅ TestResult created: ${testResult._id}`);

    // ─────────────────────────────────────────────────────────────────
    // STEP 6: Link TestResult to LabOrder
    // ─────────────────────────────────────────────────────────────────

    labOrder.testResultId = testResult._id;
    await labOrder.save();

    console.log(`[TestResult] ✅ Linked TestResult to LabOrder`);

    // ─────────────────────────────────────────────────────────────────
    // STEP 7: Update Medical Record status
    // ─────────────────────────────────────────────────────────────────

    await medicalRecordModel.MedicalRecordModel.findByIdAndUpdate(
        medicalRecord._id,
        { status: 'HAS_RESULT' }
    );

    // ─────────────────────────────────────────────────────────────────
    // STEP 8: Create audit log
    // ─────────────────────────────────────────────────────────────────

    try {
        await auditLogModel.createLog({
            userId: currentUser._id,
            action: 'CREATE_TEST_RESULT',
            entityType: 'TEST_RESULT',
            entityId: testResult._id,
            details: {
                labOrderId,
                medicalRecordId: medicalRecord._id,
                testType,
                note: `Lab tech created test result with ${testType}`,
            },
        });
    } catch (auditError) {
        console.warn('[TestResult] Audit log failed (non-blocking):', auditError.message);
    }

    return testResult;
};

// ═══════════════════════════════════════════════════════════════════════════════
// SERVICE: GET TEST RESULT DETAIL
// ═══════════════════════════════════════════════════════════════════════════════

/**
 * Get detailed test result with populated references
 * 
 * @param {String} testResultId - TestResult ID
 * @returns {Object} TestResult with populated fields
 */
const getDetail = async (testResultId) => {
    const testResult = await testResultModel.TestResultModel
        .findById(testResultId)
        .populate('labOrderId', 'sampleStatus labResultHash labTechWalletAddress')
        .populate('medicalRecordId', 'status chief_complaint diagnosis')
        .populate('patientId', 'fullName birthYear')
        .populate('createdBy', 'username email');

    if (!testResult) {
        throw new ApiError(
            StatusCodes.NOT_FOUND,
            'Không tìm thấy test result'
        );
    }

    return testResult;
};

// ═══════════════════════════════════════════════════════════════════════════════
// SERVICE: GET ALL TEST RESULTS (with optional filtering)
// ═══════════════════════════════════════════════════════════════════════════════

/**
 * Get all test results with pagination and filtering
 * 
 * @param {Object} query - Query parameters
 * @param {String} query.patientId - Filter by patient
 * @param {String} query.testType - Filter by test type
 * @param {Number} query.page - Page number (default: 1)
 * @param {Number} query.limit - Results per page (default: 10)
 * @returns {Object} Paginated test results
 */
const getAll = async (query = {}) => {
    const { patientId, testType, page = 1, limit = 10 } = query;

    // Build filter
    const filter = { _destroy: false };
    if (patientId) filter.patientId = patientId;
    if (testType) filter.testType = testType;

    const skip = (page - 1) * limit;

    const testResults = await testResultModel.TestResultModel
        .find(filter)
        .sort({ createdAt: -1 })
        .skip(skip)
        .limit(parseInt(limit))
        .populate('patientId', 'fullName')
        .populate('testType');

    const total = await testResultModel.TestResultModel.countDocuments(filter);

    return {
        data: testResults,
        pagination: {
            total,
            page: parseInt(page),
            limit: parseInt(limit),
            totalPages: Math.ceil(total / limit),
        },
    };
};

// ═══════════════════════════════════════════════════════════════════════════════
// SERVICE: UPDATE AI ANALYSIS (after FastAPI processes result)
// ═══════════════════════════════════════════════════════════════════════════════

/**
 * Update AI analysis for test result (called by FastAPI callback)
 * 
 * @param {String} testResultId - TestResult ID
 * @param {Object} aiAnalysis - New AI analysis data
 * @returns {Object} Updated TestResult
 */
const updateAiAnalysis = async (testResultId, aiAnalysis) => {
    const testResult = await testResultModel.TestResultModel.findByIdAndUpdate(
        testResultId,
        { aiAnalysis },
        { new: true, runValidators: true }
    );

    if (!testResult) {
        throw new ApiError(
            StatusCodes.NOT_FOUND,
            'Không tìm thấy test result'
        );
    }

    // Create audit log for analysis update
    try {
        await auditLogModel.createLog({
            action: 'UPDATE_TEST_RESULT_AI_ANALYSIS',
            entityType: 'TEST_RESULT',
            entityId: testResultId,
            details: {
                note: 'AI analysis updated',
                aiAnalysis,
            },
        });
    } catch (auditError) {
        console.warn('[TestResult] Audit log failed:', auditError.message);
    }

    return testResult;
};

// ═══════════════════════════════════════════════════════════════════════════════
// HELPER: Generate recommendation based on AI risk score
// ═══════════════════════════════════════════════════════════════════════════════

const generateRecommendation = (probability) => {
    if (probability >= 0.8) {
        return 'Khuyến cáo ngay lập tức: Hãy gặp bác sĩ chuyên khoa Nội tiết để được tư vấn và điều trị';
    } else if (probability >= 0.6) {
        return 'Khuyến cáo: Hãy tái kiểm tra HbA1c trong 3 tháng. Cân nhắc sử dụng Metformin';
    } else if (probability >= 0.4) {
        return 'Cảnh báo: Thay đổi lối sống (chế độ ăn, tập thể dục) và tái kiểm tra trong 6 tháng';
    } else {
        return 'Bình thường: Duy trì lối sống lành mạnh, tái kiểm tra hàng năm';
    }
};

// ═══════════════════════════════════════════════════════════════════════════════
// EXPORTS
// ═══════════════════════════════════════════════════════════════════════════════

export const testResultService = {
    createNew,
    getDetail,
    getAll,
    updateAiAnalysis,
};
```

```javascript
import { testResultService } from '~/services/testResult.service';
import { StatusCodes } from 'http-status-codes';
import ApiError from '~/utils/ApiError';

// ═══════════════════════════════════════════════════════════════════════════════
// CONTROLLER: CREATE TEST RESULT
// ═══════════════════════════════════════════════════════════════════════════════

/**
 * POST /v1/test-results/:labOrderId
 * 
 * Lab tech creates new test result with raw data
 * Optional: AI service analyzes result automatically
 * 
 * Request body:
 * {
 *   testType: 'DIABETES_TEST',
 *   rawData: { glucose: 120, ... }
 * }
 */
const createNew = async (req, res, next) => {
    try {
        const { labOrderId } = req.params;
        const result = await testResultService.createNew(labOrderId, req.body, req.user);

        res.status(StatusCodes.CREATED).json({
            message: 'Tạo test result thành công',
            data: result,
        });
    } catch (error) {
        next(error);
    }
};

// ═══════════════════════════════════════════════════════════════════════════════
// CONTROLLER: GET TEST RESULT DETAIL
// ═══════════════════════════════════════════════════════════════════════════════

/**
 * GET /v1/test-results/:testResultId
 * 
 * Get detailed view of a test result with all relationships populated
 */
const getDetail = async (req, res, next) => {
    try {
        const { testResultId } = req.params;
        const result = await testResultService.getDetail(testResultId);

        res.status(StatusCodes.OK).json({
            message: 'Lấy chi tiết test result thành công',
            data: result,
        });
    } catch (error) {
        next(error);
    }
};

// ═══════════════════════════════════════════════════════════════════════════════
// CONTROLLER: GET ALL TEST RESULTS
// ═══════════════════════════════════════════════════════════════════════════════

/**
 * GET /v1/test-results
 * 
 * Get paginated list of test results with optional filtering
 * 
 * Query parameters:
 * - patientId: Filter by patient
 * - testType: Filter by test type (DIABETES_TEST, DNA_TEST, etc.)
 * - page: Page number (default: 1)
 * - limit: Results per page (default: 10)
 */
const getAll = async (req, res, next) => {
    try {
        const result = await testResultService.getAll(req.query);

        res.status(StatusCodes.OK).json({
            message: 'Lấy danh sách test results thành công',
            data: result,
        });
    } catch (error) {
        next(error);
    }
};

// ═══════════════════════════════════════════════════════════════════════════════
// CONTROLLER: UPDATE AI ANALYSIS (FastAPI Callback)
// ═══════════════════════════════════════════════════════════════════════════════

/**
 * PATCH /v1/test-results/:testResultId/ai-analysis
 * 
 * Update AI analysis for test result (called by FastAPI after processing)
 * 
 * Request body:
 * {
 *   aiAnalysis: { diabetes: { riskScore: 0.85, ... } }
 * }
 */
const updateAiAnalysis = async (req, res, next) => {
    try {
        const { testResultId } = req.params;
        const { aiAnalysis } = req.body;

        if (!aiAnalysis) {
            throw new ApiError(
                StatusCodes.BAD_REQUEST,
                'aiAnalysis field is required'
            );
        }

        const result = await testResultService.updateAiAnalysis(testResultId, aiAnalysis);

        res.status(StatusCodes.OK).json({
            message: 'Cập nhật AI analysis thành công',
            data: result,
        });
    } catch (error) {
        next(error);
    }
};

// ═══════════════════════════════════════════════════════════════════════════════
// EXPORTS
// ═══════════════════════════════════════════════════════════════════════════════

export const testResultController = {
    createNew,
    getDetail,
    getAll,
    updateAiAnalysis,
};
```
